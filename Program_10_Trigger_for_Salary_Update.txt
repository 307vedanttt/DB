CREATE TABLE Employee(
    EmpID INT PRIMARY KEY,
    Name VARCHAR(50),
    Salary INT
);

CREATE TABLE Employee_Audit(
    EmpID INT,
    OldSalary INT,
    NewSalary INT
);

DELIMITER //

CREATE TRIGGER Salary_Update
AFTER UPDATE ON Employee
FOR EACH ROW
BEGIN
    INSERT INTO Employee_Audit
    VALUES(OLD.EmpID, OLD.Salary, NEW.Salary);
END //

DELIMITER ;
