CREATE TABLE Customer(
    CustomerID INT PRIMARY KEY,
    Name VARCHAR(50)
);

CREATE TABLE Orders(
    OrderID INT PRIMARY KEY,
    CustomerID INT,
    Amount INT,
    FOREIGN KEY (CustomerID) REFERENCES Customer(CustomerID)
);
